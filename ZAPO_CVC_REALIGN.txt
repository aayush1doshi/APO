        go_sender =  cl_cam_address_bcs=>create_internet_address( gc_sndr_mail ).

        go_send_request->set_sender( i_sender =  go_sender ).

*        setting recipents of the mail
        LOOP AT gt_recipents INTO lw_recipents.
          lv_mail = lw_recipents-smtp_addr.

          gi_recipient = cl_cam_address_bcs=>create_internet_address( lv_mail ).

          go_send_request->add_recipient(
            EXPORTING
              i_recipient = gi_recipient
              i_express    = abap_true
          ).

          go_send_request->set_status_attributes( gc_req_attr_n ).

        ENDLOOP.

*        setting CC mail address
        LOOP AT gt_zapoparam ASSIGNING <lfs_zapoparam> WHERE param3 = gc_param3_mail AND param4 = gc_param4_cc.
          lv_mail = <lfs_zapoparam>-value1.
          cl_cam_address_bcs=>create_internet_address(
            EXPORTING
              i_address_string = lv_mail    " Internet Mail (SMTP) Address
            RECEIVING
              result           =  gi_recipient   " Reference to BAS Address
          ).

          go_send_request->add_recipient(
            EXPORTING
              i_recipient  = gi_recipient    " Recipient of Message
              i_copy       = abap_true    " Send Copy
          ).
        ENDLOOP.
        "madhvi
*        reading mail subject & body
        CALL FUNCTION 'READ_TEXT'
          EXPORTING
            client                  = sy-mandt
            id                      = gc_tdid_st
            language                = sy-langu
            name                    = 'ZAPO_CVC_MAIL_ERR_BODY'
            object                  = gc_tdobject_text
          TABLES
            lines                   = gt_lines
          EXCEPTIONS
            id                      = 1
            language                = 2
            name                    = 3
            not_found               = 4
            object                  = 5
            reference_check         = 6
            wrong_access_to_archive = 7
            OTHERS                  = 8.
        IF sy-subrc <> 0.
          MESSAGE ID sy-msgid TYPE gc_suc NUMBER sy-msgno
            WITH sy-msgv1 sy-msgv2 sy-msgv3 sy-msgv4 DISPLAY LIKE gc_e.
          RETURN.
        ENDIF.

*        reading mail subject
        READ TABLE gt_lines ASSIGNING <lfs_line> INDEX 1.
        IF sy-subrc = 0.
          lv_subj = <lfs_line>-tdline.
          REPLACE '[BSG]' IN lv_subj WITH p_bsggrp.
        ENDIF.

*        deleting the mail subject
        DELETE gt_lines INDEX 1.

*        converting tlines to mail body
        LOOP AT gt_lines ASSIGNING <lfs_line>.
          lv_tabix = sy-tabix.
          APPEND INITIAL LINE TO gt_mail_body ASSIGNING <lfs_mail_body>.
          IF <lfs_mail_body> IS ASSIGNED.
            CASE lv_tabix.
              WHEN 2.
                CONCATENATE <lfs_line>-tdline p_mod INTO <lfs_mail_body>-line SEPARATED BY space.
              WHEN 3.
*        reading Business group text name
                UNASSIGN <lfs_zapoparam>.
                READ TABLE gt_zapoparam ASSIGNING <lfs_zapoparam>
                  WITH KEY param3 = gc_param3_mpos param4 = p_bsggrp
                  BINARY SEARCH.
                IF sy-subrc = 0.
                  lv_bsg_text = <lfs_zapoparam>-remarks.
                ELSE.
                  lv_bsg_text = p_bsggrp.
                ENDIF.

                CONCATENATE <lfs_line>-tdline lv_bsg_text INTO <lfs_mail_body>-line SEPARATED BY space.
              WHEN OTHERS.
                <lfs_mail_body>-line = <lfs_line>-tdline.
            ENDCASE.
          ENDIF.
          IF lv_tabix = 7.
            LOOP AT gt_rulelog ASSIGNING <lfs_rulelog>.
              APPEND INITIAL LINE TO gt_mail_body ASSIGNING <lfs_mail_body>.
              IF <lfs_mail_body> IS ASSIGNED.
                CONCATENATE gc_begin_row gc_begin_cell <lfs_rulelog>-rule gc_end_cell gc_begin_cell
                    <lfs_rulelog>-stat gc_end_cell gc_begin_cell <lfs_rulelog>-err_desc gc_end_cell gc_end_row
                  INTO <lfs_mail_body>-line.
              ENDIF.
            ENDLOOP.
          ENDIF.
        ENDLOOP.

        "Add document
        go_document =  cl_document_bcs=>create_document(

                           i_text        = gt_mail_body ).

        CALL METHOD go_send_request->send( ).
        IF sy-subrc EQ 0.
          COMMIT WORK.
          MESSAGE 'Realignment Error: Log sent through Mail'(014) TYPE gc_suc DISPLAY LIKE gc_e.
        ENDIF.

  "! updating the status in the ZAPO_IBP_STATUS table
  METHOD upd_ibp_status.
    DATA:
      lv_name_text TYPE ad_namtext.

*    getting the full name of user
    SELECT SINGLE name_text
      FROM v_usr_name
      INTO lv_name_text
      WHERE bname = sy-uname.
    UNASSIGN <gfs_zapoparam>.
    READ TABLE gt_zapoparam ASSIGNING <gfs_zapoparam>
      WITH KEY
        param3 = gc_param3_seq param4 = p_bsggrp.
    IF sy-subrc = 0 AND <gfs_zapoparam> IS ASSIGNED.
      CLEAR gw_ibp_status.
      gw_ibp_status-mandt = sy-mandt.
      gw_ibp_status-sequence =  <gfs_zapoparam>-value1.
      gw_ibp_status-activity = <gfs_zapoparam>-value2.
      gw_ibp_status-business = p_bsggrp.
      CONCATENATE sy-datum+4(2) sy-datum(4) INTO gw_ibp_status-month_b SEPARATED BY '.'.
      gw_ibp_status-date_b = sy-datum.
      gw_ibp_status-date_b = sy-datum.
      gw_ibp_status-userid = sy-uname.
      gw_ibp_status-username = lv_name_text.
      gw_ibp_status-status = iv_stat.
      IF gw_ibp_status-status IS INITIAL.
*        if the status is error, then update the remarks col with the failed rule name
        CONCATENATE 'Rule:'(023) iv_rule 'Failed to Execute'(024) INTO gw_ibp_status-remarks SEPARATED BY space.
      ENDIF.
      SELECT SINGLE sequence activity business month_b
        FROM zapo_ibp_status
        INTO CORRESPONDING FIELDS OF gw_ibp_status
        WHERE sequence = gw_ibp_status-sequence AND activity = gw_ibp_status-activity AND
          business = gw_ibp_status-business AND month_b = gw_ibp_status-month_b.
      IF  sy-subrc = 0 .
        CALL FUNCTION 'ENQUEUE_EZAPO_IBPSTAT'
          EXPORTING
            mode_zapo_ibp_status = 'E'
            mandt                = sy-mandt
            sequence             = gw_ibp_status-sequence
            activity             = gw_ibp_status-activity
            business             = gw_ibp_status-business
            month_b              = gw_ibp_status-month_b
          EXCEPTIONS
            foreign_lock         = 1
            system_failure       = 2
            OTHERS               = 3.
        IF sy-subrc <> 0.
          MESSAGE ID sy-msgid TYPE gc_suc NUMBER sy-msgno
            WITH sy-msgv1 sy-msgv2 sy-msgv3 sy-msgv4
            DISPLAY LIKE gc_e.
          LEAVE LIST-PROCESSING.
        ENDIF.
        gw_ibp_status-date_b = sy-datum.
        gw_ibp_status-date_b = sy-datum.
        gw_ibp_status-userid = sy-uname.
        gw_ibp_status-username = lv_name_text.
        MODIFY zapo_ibp_status FROM gw_ibp_status.
        CALL FUNCTION 'DEQUEUE_EZAPO_IBPSTAT'
          EXPORTING
            mode_zapo_ibp_status = 'E'
            mandt                = sy-mandt
            sequence             = gw_ibp_status-sequence
            activity             = gw_ibp_status-activity
            business             = gw_ibp_status-business
            month_b              = gw_ibp_status-month_b.
      ELSE.
        INSERT zapo_ibp_status FROM gw_ibp_status.
      ENDIF.
      IF gw_ibp_status-status IS INITIAL.
        MESSAGE 'Realignment Error: Log sent through Mail'(014) TYPE gc_suc DISPLAY LIKE gc_e.
      ELSE.
        IF gv_flg_norlgn = abap_false.
          MESSAGE 'No Realignment needed, Master Data is correct'(026) TYPE gc_suc.
        ELSE.
          MESSAGE 'Realignment & IBP Status Updated Sucessfully'(025) TYPE gc_suc.

  METHOD list_of_inputs.
    IF  ( gv_old_bg IS INITIAL OR gv_old_bg <> p_bsggrp ) AND p_bsggrp IS NOT INITIAL.
      gv_old_bg = p_bsggrp.
      IF p_bsggrp IS NOT INITIAL.
        CLEAR : lt_apoparam[],gt_value[], gt_zapoparm[].
        SELECT param1
           param2
           param3
           param4
           param5
           value1
           value2
           remarks
          FROM zapoparam INTO TABLE lt_apoparam
          WHERE param1 = p_mod AND
                param2 = 'ZAPO_CVCREALIGN' AND
                param3 = p_bsggrp AND param4 = 'RULE'.
        IF sy-subrc IS INITIAL.
          SORT lt_apoparam BY param5.
        ENDIF.

        LOOP AT lt_apoparam ASSIGNING <lfs_apoparam>.
          UNASSIGN <gfs_value>.
          APPEND INITIAL LINE TO gt_value ASSIGNING <gfs_value>.
          IF <gfs_value> IS ASSIGNED.
            <lfs_apoparam>-rule =   <lfs_apoparam>-param4 && '-' && <lfs_apoparam>-param5 .
            <lfs_apoparam>-rule_desc = <lfs_apoparam>-remarks.
            TRANSLATE <lfs_apoparam>-rule TO UPPER CASE.
            TRANSLATE <lfs_apoparam>-rule_desc TO UPPER CASE.
            <gfs_value>-key = <lfs_apoparam>-rule.
            <gfs_value>-text = <lfs_apoparam>-rule_desc.
            gws_zapoparm-rule = <lfs_apoparam>-rule.
            gws_zapoparm-remarks = <lfs_apoparam>-remarks.
            APPEND gws_zapoparm TO gt_zapoparm.
            CLEAR gws_zapoparm.
          ENDIF.
        ENDLOOP.
        UNASSIGN <gfs_value>.
        SORT gt_value BY key .
        DELETE ADJACENT DUPLICATES FROM gt_value COMPARING key.
      ENDIF.
      IF p_rule_s IS NOT INITIAL.
        READ TABLE gt_value WITH KEY key = p_rule_s TRANSPORTING NO FIELDS.
        IF sy-subrc IS NOT INITIAL.
          CLEAR p_rule_s.
        ENDIF.
      ENDIF.
*    CHECK NOT gt_value[] IS INITIAL.
*     clear : p_rule_s.
      CALL FUNCTION 'VRM_SET_VALUES'
        EXPORTING
          id              = 'P_RULE_S'
          values          = gt_value[]
        EXCEPTIONS
          id_illegal_name = 1
          OTHERS          = 2.

      IF sy-subrc = 0.
*      CLEAR P_RULE_S.
      ENDIF.
    ENDIF.
  ENDMETHOD.                    "LIST_OF_INPUTS
