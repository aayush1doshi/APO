         param2
         param3
         param4

      ENDIF.

  FROM zapoparam
  INTO TABLE lt_zapoparam
  WHERE  param1 = lc_param1
  AND param2 = lc_param2

    CLEAR: lw_zapoparam.

                                            AND  param5 = lc_strgr.
      lw_strgr-strgr = lw_zapoparam-value2.
      lw_strgr-value = lw_zapoparam-value1.
      APPEND lw_strgr TO lt_strgr.
      CLEAR: lw_zapoparam,lw_strgr.

    CLEAR: lw_zapoparam.
    LOOP AT lt_zapoparam INTO lw_zapoparam WHERE param4 = lc_param4

      lw_mtvfp-mtvfp = lw_zapoparam-value2.
      lw_mtvfp-value = lw_zapoparam-value1.
      APPEND lw_mtvfp TO lt_mtvfp.
      CLEAR: lw_zapoparam,lw_mtvfp.

  CLEAR: lw_product.
  LOOP AT it_product INTO lw_product.
    lw_matnr-matnr = lw_product-matnr.
    lw_matnr-sign = lw_product-sign.

    lw_matnr2-option = lw_product-option.
    lw_matnr2-low = lw_product-matnr.
    APPEND lw_matnr2 TO lr_matnr.

    CLEAR: lw_product.
  ENDLOOP.

  DELETE lt_matnr WHERE matnr NOT IN lr_matnr.

  CLEAR: lw_loc.
  LOOP AT it_location INTO lw_location.
    lw_loc-location = lw_location-location.
    APPEND lw_loc TO lt_loc.
    CLEAR: lw_loc.

    lw_location1-sign = lw_location-sign.
    lw_location1-option = lw_location-option.
    lw_location1-low = lw_location-location.

  CALL FUNCTION 'Z_SCM_REDEP_FETCH_PAG' DESTINATION lw_rfc-rfcdest  "Read PAG CVC and Quantity
    EXPORTING
      i_div     = i_div
      it_matnr  = lt_matnr
      it_locno  = lt_loc
      it_strgr  = lt_strgr
      it_mtvpf  = lt_mtvfp
    IMPORTING
      et_matloc = lt_matloc_d
      et_return = lt_return.

  IF lt_matloc_d IS NOT INITIAL.

    DESCRIBE TABLE lt_matloc_d LINES gv_lines.
    IF gv_lines GT 98.
      gv_cnt = gv_lines / 50.
      gv_cnt = gv_cnt + 1.
      gv_flag = abap_true.
    ENDIF.

    CLEAR : gv_logsys.
*** logical system
    CALL FUNCTION '/SAPAPO/GET_OWN_LOGSYS'                        " Determine the Separate Logical System
      IMPORTING
        ev_ownlogsys = gv_logsys.

    CLEAR: lv_end_time, lv_start_time,lv_dt_from, lv_dt_to.

    lv_dt_to = i_to_date.
    CONCATENATE lv_dt_from '000000' INTO lv_dt_from.
    CONCATENATE lv_dt_to '235959' INTO lv_dt_to.

    lv_start_time = lv_dt_from.
    lv_end_time = lv_dt_to.

        lw_product_selection-sign = 'I'.
        lw_product_selection-option = 'EQ'.
        lw_product_selection-low = lw_matloc_d-matnr.
        APPEND lw_product_selection TO lt_product_selection.

        lw_location_selection-option = 'EQ'.
        lw_location_selection-low = lw_matloc_d-werks.
        APPEND lw_location_selection TO lt_location_selection.
        CLEAR: lw_location_selection.
        CLEAR: lw_matloc_d.
      ENDLOOP.

        lw_atpcat_selection-atpcat = lw_zapoparam-value1.
        APPEND lw_atpcat_selection TO lt_atpcat_selection.
        CLEAR: lw_atpcat_selection.
        CLEAR: lw_zapoparam.

      DESCRIBE TABLE lt_location_selection LINES lv_cnt2.

      DESCRIBE TABLE lt_atpcat_selection LINES lv_cnt3.

      lv_cnt = lv_cnt1 + lv_cnt2 + lv_cnt3.

      lw_extension_in-structure = 'MAX_RANGE_ENTRIES'.

      lw_exclude-no_charact_req_input_flag =
      lw_exclude-no_documents_flag =

        EXPORTING
          logical_system              = gv_logsys

          end_time                    = lv_end_time
         exclude_export_flags        = lw_exclude

          plng_version                = '000'

          requirements                = lt_requirements

         return                      = lt_return1

                .



      IF lt_return1 IS NOT INITIAL.

        CLEAR: lw_return.
        LOOP AT lt_return1 INTO lw_return .
          APPEND lw_return TO et_return.
          CLEAR: lw_return.
        ENDLOOP.
      ENDIF.

        IF it_str_loc IS NOT INITIAL.

          LOOP AT it_str_loc INTO lw_str_loc.
            CLEAR: lw_lgort.
            lw_lgort-sign = lw_str_loc-sign.

        ENDIF.

    ELSE.

        CASE lv_index.
          WHEN '1'.

            lv_no_to = 48.
          WHEN OTHERS.
            lv_no_from = lv_no_to + 1.

          lw_product_selection-sign = 'I'.
          lw_product_selection-option = 'EQ'.
          lw_product_selection-low = lw_matloc_d-matnr.
          APPEND lw_product_selection TO lt_product_selection.

          lw_location_selection-option = 'EQ'.
          lw_location_selection-low = lw_matloc_d-werks.

          CLEAR: lw_matloc_d.

        DESCRIBE TABLE lt_location_selection LINES lv_cnt2.


        CLEAR lv_cnt3.
        DESCRIBE TABLE lt_atpcat_selection LINES lv_cnt3.

        lw_extension_in-valuepart1 = lv_cnt."lv_cnt_f.
        APPEND lw_extension_in TO lt_extension_in.
        CLEAR: lw_extension_in.

        lw_exclude-no_peg_input_flag =
        lw_exclude-no_charact_req_input_flag =
        lw_exclude-no_documents_flag =
        lw_exclude-no_receipts_flag =
        lw_exclude-no_peg_output_flag =
        lw_exclude-no_cfg_headers =

           exclude_export_flags        = lw_exclude

*     TRANSFER_MODE               =
            plng_version                = '000'
            io_node_selection           = 'I'
*   IMPORTING
*     NUMBER_OF_ORDERS            =
         TABLES
            product_selection           = lt_product_selection
            location_selection          = lt_location_selection
*     LOCTYPE_SELECTION           =
*     PRODUCT_LOCATION_KEYS       =
*     ORDER_SELECTION             =
*     ORDERTYPE_SELECTION         =
*     ORDER_GUIDS                 =
*     ORDER_KEYS                  =
            atpcat_selection            = lt_atpcat_selection
*     CUSTOMER_SELECTION          =
*     CFG_SELECTION               =
            requirements                = lt_requirements
*     RECEIPTS                    =
*     PEGGING_INPUT_NODES         =
*     PEGGING_OUTPUT_NODES        =
*     CFG_HEADERS                 =
*     CFG_INSTANCES               =
*     CFG_PART_OF                 =
*     CFG_VALUES                  =
*     CFG_VAR_KEYS                =
*     CFG_BLOB                    =
           return                      = lt_return1
           extension_in                = lt_extension_in
           extension_out               = lt_extension_out
                  .



        IF lt_return1 IS NOT INITIAL.

          CLEAR: lw_return.
          LOOP AT lt_return1 INTO lw_return .
            APPEND lw_return TO et_return.
            CLEAR: lw_return.
          ENDLOOP.
